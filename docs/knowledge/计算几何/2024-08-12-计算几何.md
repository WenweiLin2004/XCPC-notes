```cpp
    Point get_line_intersection(Point p,Vector v,Point q,Vector w){//两直线的交点
        //使用前提，直线必须有交点
        //cross(v, w) == 0则两直线平行或者重合
        if(!sign(cross(v,w))) return {INF,INF};
        Vector u=p-q;
        double t=cross(w,u)/cross(v,w);
        Point o=p+v*t;
        auto on_segment=[&](Point p,Point a,Point b){
            return sign(dot(p-a,p-b))<=0;
        };
        //若判断线段/射线是否相交 还需判断交点是否在线段/射线上
        if(!on_segment(o,p,p+v)||!on_segment(o,q,q+w))return {INF,INF};
        return o;

            res=max({res,(d-hull[j]).dis(0),(e-hull[j]).dis(0)});
        }
        return res;
    }

            auto d=hull[i],e=hull[i+1];

            while(cmp(get_project(d,e,hull[b]),get_project(d,e,hull[b+1]))<0)b=(b+1)%n;
            if(!i)c=a;
            while(cmp(get_project(d,e,hull[c]),get_project(d,e,hull[c+1]))>0)c=(c+1)%n;
            auto x=hull[a],y=hull[b],z=hull[c];
            auto h=area(d,e,x)/(e-d).length();
            auto w=dot(y-z,e-d)/(e-d).length();
            if(h*w<min_area){
                min_area=h*w;
                ans[0] = d + (e - d).norm() * get_project(d, e, y);
                ans[3] = d + (e - d).norm() * get_project(d, e, z);
                auto u = (rotate(e - d, -PI / 2)).norm();
                ans[1] = ans[0] + u * h;
                ans[2] = ans[3] + u * h;            
            }

        sort(line.begin(),line.end(),line_cmp);//对每条直线按照极坐标进行排序
        int hh=0,tt=-1;//定义双端队列对头和队尾元素

            //如果line[l[tt-1]]和line[l[tt]]的交点在line[i]的右侧,删除队尾元素

            //如果line[l[hh]]和line[l[hh+1]]的交点在line[i]的右侧,删除队头元素

    double line_area(double a, int side,vector<vector<Point>> &tr){
        vector<Point> q;
        for (int i = 0; i < tr.size(); i ++ ){
            auto t = tr[i];
            if (cmp(t[0].x, a) > 0 || cmp(t[2].x, a) < 0) continue;
            if (!cmp(t[0].x, a) && !cmp(t[1].x, a)){
                if (side) q.emplace_back(t[0].y, t[1].y);
            }
            else if (!cmp(t[2].x, a) && !cmp(t[1].x, a)){
                if (!side) q.emplace_back(t[2].y, t[1].y);
            }
            else{
                static double d[3];
                int u = 0;
                for (int j = 0; j < 3; j ++ ){
                    auto o = get_line_intersection(t[j], t[(j + 1) % 3] - t[j], {a, -INF}, {0,INF * 2});
                    if (cmp(o.x, INF))d[u++] = o.y;
                }
                if (u){
                    sort(d, d + u);
                    q.emplace_back(d[0], d[u - 1]);
                }
            }
        }
        if(!q.size())return 0;
        for (int i = 0; i < q.size(); i ++ ){
            if (q[i].x > q[i].y)swap(q[i].x, q[i].y);
        }
        sort(q.begin(),q.end());
        double res = 0, st = q[0].x, ed = q[0].y;
        for (int i = 1; i < q.size(); i ++ ){
            if (q[i].x <= ed) ed = max(ed, q[i].y);
            else{
                res += ed - st;
                st = q[i].x, ed = q[i].y;
            }
        }
        res += ed - st;
        return res;
    }
```

### Andrew算法
* x，y从小到大排序
* 正向做一遍，反向做一遍，用栈维护
* 如果栈顶向量与当前点围成的面积为负，则弹出
* 凸包的点为逆时针

```cpp
int stk[N],top;
bool used[N];
vector<Point> andrew(vector<Point> &q){
	sort(q.begin(),q.end());
	int n=q.size();
    top=0;
    for(int i=0;i<n;i++)used[i]=false;
    for(int i=0;i<n;i++){
        while(top>=2&&sign(area(q[stk[top-1]],q[stk[top]],q[i]))<=0){
            if(sign(area(q[stk[top-1]],q[stk[top]],q[i])<0))used[stk[top--]]=false;
            else top--;
        }
        stk[++top]=i;
        used[i]=true;
    }
    used[0]=false;
    for(int i=n-1;i>=0;i--){
        if(used[i])continue;
        while(top>=2&&sign(area(q[stk[top-1]],q[stk[top]],q[i]))<=0)top--;
        stk[++top]=i;
    }
	vector<Point> hull;
	for(int i=1;i<top;i++)hull.push_back(q[stk[i]]);
	return hull;
}
```

```cpp
double NearestPointDis(int l,int r,vector<Point> &a){
    if(l>=r)return LLONG_MAX;
    if(l+1==r){
        if(a[l].y>a[r].y)swap(a[l],a[r]);
        return (a[l]-a[r]).dis();
    }
    int mid=l+r>>1;
    int t=a[mid].x;
    double d=min(NearestPointDis(l,mid,a),NearestPointDis(mid+1,r,a));
    Merge(l,r,a);
    vector<int> pos;
    for(int i=l;i<=r;i++){
        if(1ll*(a[i].x-t)*(a[i].x-t)<d)pos.push_back(i);
    }
    for(int i=0;i<pos.size();i++){
        for(int j=i+1;j<pos.size()&&1ll*(a[pos[i]].y-a[pos[j]].y)*(a[pos[i]].y-a[pos[j]].y)<d;j++){
            d=min(d,(a[pos[i]]-a[pos[j]]).dis());
        }
    }
    return d;
```

```cpp
        auto t = tr[i];
        if (cmp(t[0].x, a) > 0 || cmp(t[2].x, a) < 0) continue;
        if (!cmp(t[0].x, a) && !cmp(t[1].x, a)){
            if (side) q.emplace_back(t[0].y, t[1].y);
        }
        else if (!cmp(t[2].x, a) && !cmp(t[1].x, a)){
            if (!side) q.emplace_back(t[2].y, t[1].y);
        }
        else{
            static double d[3];
            int u = 0;
            for (int j = 0; j < 3; j ++ ){
                auto o = get_line_intersection(t[j], t[(j + 1) % 3] - t[j], {a, -INF}, {0,INF * 2});
                if (cmp(o.x, INF))d[u++] = o.y;
            }
            if (u){
                sort(d, d + u);
                q.emplace_back(d[0], d[u - 1]);
            }
        }
    }
    if(!q.size())return 0;
    for (int i = 0; i < q.size(); i ++ ){
        if (q[i].x > q[i].y)swap(q[i].x, q[i].y);
    }
    sort(q.begin(),q.end());
    double res = 0, st = q[0].x, ed = q[0].y;
    for (int i = 1; i < q.size(); i ++ ){
        if (q[i].x <= ed) ed = max(ed, q[i].y);
        else{
            res += ed - st;
            st = q[i].x, ed = q[i].y;
        }
    }
    res += ed - st;
    return res;
}
double range_area(double a, double b,vector<vector<Point>> &tr){
    return (line_area(a, 1, tr) + line_area(b, 0, tr)) * (b - a) / 2;
}
double triangle_union_area(vector<vector<Point>> &tr){
    vector<double> xs;
    for(int i=0;i<tr.size();i++){
        for(int j=0;j<3;j++){
            xs.push_back(tr[i][j].x);
        }
        sort(tr[i].begin(),tr[i].begin()+3);
    }
    for(int i=0;i<tr.size();i++){
        for(int j=i+1;j<tr.size();j++){
            for(int x=0;x<3;x++){
                for(int y=0;y<3;y++){
                    auto o=get_line_intersection(tr[i][x],tr[i][(x+1)%3]-tr[i][x],tr[j][y],tr[j][(y+1)%3]-tr[j][y]);
                    if(cmp(o.x,INF))xs.push_back(o.x);
                }
            }
        }
    }
    sort(xs.begin(),xs.end());
    double res=0;
    for(int i=0;i+1<xs.size();i++){
        if(cmp(xs[i],xs[i+1]))res+=range_area(xs[i],xs[i+1],tr);
    }
    return res;
}
```

## 最小圆覆盖
```cpp
struct Circle{
    Point p;
    double r;
};

pair<Point,Point> get_line(Point a,Point b){
    //找到与向量ab垂直的向量的起点与终点
    return {(a+b)/2,rotate(b-a,PI/2)};
}

Circle get_circle(Point a,Point b,Point c){
    //根据不共线的三点求出圆的圆心坐标和半径
    auto u=get_line(a,b),v=get_line(a,c);//求出垂直于ab和ac的向量
    auto get_line_intersection=[&](Point p,Vector v,Point q,Vector w){
        Vector u=p-q;
        double t=cross(w,u)/cross(v,w);
        return p+v*t;
    };
    auto p=get_line_intersection(u.first,u.second,v.first,v.second);//两向量交点即为所求圆的圆心
    return {p,(p-a).length()};
}
//----------------求最小圆覆盖,期望O(n)---------------------
Circle smallest_circle(vector<Point> &po){
    random_shuffle(po.begin(),po.end());
    Circle c={po[0],0};//最开始时直接将第一个点看作圆心，半径为0
    for(int i=1;i<po.size();i++){
        if(cmp(c.r,(c.p-po[i]).length())<0){
            //如果第i个点不在前i-1个点的最小圆覆盖内，则进入下一重循环
            c={po[i],0};//将第i个点当作圆心，半径为0，重新确定圆
            for(int j=0;j<i;j++){
                if(cmp(c.r,(c.p-po[j]).length())<0){
                    c={(po[i]+po[j])/2,(po[i]-po[j]).length()/2};
                    for(int k=0;k<j;k++){
                        if(cmp(c.r,(c.p-po[k]).length())<0){//确定了最小圆覆盖上的三个点，直接求出圆
                            c=get_circle(po[i],po[j],po[k]);
                        }
                    }
                }
            }
        }
    }
    return c;
}
```

## 自适应辛普森积分求圆的面积并
```cpp
const double eps=1e-12;
int n;
struct Point{
    double x,y;
    bool operator<(const Point &t)const{
        return x!=t.x?x<t.x:y<t.y;
    }
};
struct Circle{
    Point p;
    double R;
} c[N];
Point q[N];
int dcmp(double x,double y){
    if(fabs(x-y)<eps)return 0;
    if(x<y)return -1;
    return 1;
}
double f(double x){
    int cnt=0;
    for(int i=0;i<n;i++){
        double X=fabs(x-c[i].p.x),R=c[i].R;
        if(dcmp(X,R)<0){
            double Y=sqrt(R*R-X*X);
            q[cnt++]={c[i].p.y-Y,c[i].p.y+Y};
        }
    }
    if(!cnt)return 0;
    sort(q,q+cnt);
    double res=0,st=q[0].x,ed=q[0].y;
    for(int i=1;i<cnt;i++){
        if(q[i].x<=ed)ed=max(ed,q[i].y);
        else{
            res+=ed-st;
            st=q[i].x,ed=q[i].y;
        }
    }
    return res+ed-st;
}

double simpson(double l,double r,double sl,double sr){
    double mid=(l+r)/2;
    return (r-l)*(sl+4*f(mid)+sr)/6;
}
double asr(double l,double r,double s,double sl,double sr){
    double mid=(l+r)/2;
    double S=f(mid);
    double left=simpson(l,mid,sl,S),right=simpson(mid,r,S,sr);
    if(fabs(left+right-s)<eps)return left+right;
    return asr(l,mid,left,sl,S)+asr(mid,r,right,S,sr);
}
double dis(Point a,Point b){
    return sqrt((a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y));
}
void init(){		//预处理，删除被包含的圆
    int cnt = 0;
    for (int i = 0, j; i < n; ++i) {
        for (j = i+1; j < n; ++j) {
            if(c[i].R-c[j].R + dis(c[i].p, c[j].p) < eps){
                break;
            }
        }
        if(j >= n) c[cnt++] = c[i];
    }
    n = cnt;
}
vector<Point> segs;
void merge(){
    sort(segs.begin(),segs.end());
    vector<Point> res;
    double st=segs[0].x,ed=segs[0].y;
    for(int i=1;i<segs.size();i++){
        if(dcmp(segs[i].x,ed)<=0)ed=max(ed,segs[i].y);
        else{
            res.push_back({st,ed});
            st=segs[i].x,ed=segs[i].y;
        }
    }
    res.push_back({st,ed});
    segs.swap(res);
}
void solve(){
    cin>>n;
    for(int i=0;i<n;i++)cin>>c[i].p.x>>c[i].p.y>>c[i].R,segs.push_back({c[i].p.x - c[i].R, c[i].p.x + c[i].R});
    sort(c,c+n,[&](Circle a,Circle b){
        return a.R<b.R;
    });
    init();
    merge();
    double res=0;
    for(int i=0;i<segs.size();i++){
        auto [l,r]=segs[i];
        res+=asr(l,r,f((l+r)/2),f(l),f(r));
    }
    cout<<setprecision(3)<<fixed<<res<<"\n";
}
```
