# 多项式

## cdq分治FFT

给定序列 $g_{1\dots n - 1}$，求序列 $f_{0\dots n - 1}$。

其中 $f_i=\sum_{j=1}^if_{i-j}g_j$，边界为 $f_0=1$。

答案对 $998244353$ 取模。

考虑递归计算计算$f_l$到$f_{mid}$，接着计算后半部分贡献
$$
w_x =\sum_{i=l}^{mid}f_i \times g_{x-i}\\
w_x=\sum_{i=l}^{x-1}f_i \times g_{x-i}\\
w_x=\sum_{i=0}^{x-l-1}f_{i+l} \times g_{x-i-l}
设a_i=f_{i+l},b_i=g_{i+1}
w_x=\sum_{i=0}^{x-l-1}a_i \times b_{x-l-1-i}
$$

```cpp
namespace NTT{
    static constexpr int P=998244353;
    int rev[N],A[N],B[N],bit=0,tot,g=3,gi;
    int qpow(int a,int b){
        int ans=1;
        for(;b;b>>=1){
            if(b&1)ans=ans*a%P;
            a=a*a%P;
        }
        return ans;
    }
    int inv(int n){
        return qpow(n,P-2);
    }
    void ntt(int *a,int inv){
        for(int i=0;i<tot;i++){
            if(i<rev[i])swap(a[i],a[rev[i]]);
        }
        for(int mid=1;mid<tot;mid<<=1){
            int g1=qpow(inv==1?g:gi,(P-1)/(mid*2));
            for(int i=0;i<tot;i+=mid*2){
                int gk=1;
                for(int j=0;j<mid;j++,gk=gk*g1%P){
                    auto x=a[i+j],y=gk*a[i+j+mid]%P;
                    a[i+j]=(x+y)%P,a[i+j+mid]=(x-y+P)%P;
                }
            }
        }
        if(inv==-1){
            int tmp=NTT::inv(tot);
            for(int i=0;i<tot;i++)a[i]=a[i]*tmp%P;
        }
    }
    void init(int n){
        bit=0;
        while((1<<bit)<n)bit++;
        tot=1<<bit;
        for(int i=0;i<tot;i++)rev[i]=(rev[i>>1]>>1)|((i&1)<<(bit-1));
        gi=inv(g);
    }
    void mul(int *a,int *b){
        ntt(a,1),ntt(b,1);
        for(int i=0;i<tot;i++)a[i]=a[i]*b[i]%P;
        ntt(a,-1);
    }
    void cdq_ntt(int *a,int *b,int l,int r){
        if(l==r)return;
        int mid=l+r>>1,len=r-l+1;
        cdq_ntt(a,b,l,mid);
        init(len);
        for(int i=l;i<=mid;i++)A[i-l]=a[i];
        for(int i=1;i<=r-l;i++)B[i-1]=b[i];
        mul(A,B);
        for(int i=mid+1;i<=r;i++)(a[i]+=A[i-l-1])%=P;
        for(int i=0;i<tot;i++)A[i]=B[i]=0;
        cdq_ntt(a,b,mid+1,r);
    }
}
int f[N],g[N];
void solve(){
    int n;
    cin>>n;
    f[0]=1;
    for(int i=1;i<n;i++)cin>>g[i];
    NTT::cdq_ntt(f,g,0,n-1);
    for(int i=0;i<n;i++)cout<<f[i]<<" \n"[i==n-1];
}
```

```cpp
namespace Poly{
    using poly=vector<int>;
    vector<int> rev;
    const int P=998244353;
    int qpow(int a,int b){
        int ans=1;
        for(;b;b>>=1){
            if(b&1)ans=ans*a%P;
            a=a*a%P;
        }
        return ans;
    }
    int bit=0,tot,g=3,gi=qpow(g,P-2);
    void ntt(poly &a,int inv){
        for(int i=0;i<tot;i++){
            if(i<rev[i])swap(a[i],a[rev[i]]);
        }
        for(int mid=1;mid<tot;mid<<=1){
            int g1=qpow(inv==1?g:gi,(P-1)/(mid*2));
            for(int i=0;i<tot;i+=mid*2){
                int gk=1;
                for(int j=0;j<mid;j++,gk=gk*g1%P){
                    auto x=a[i+j],y=gk*a[i+j+mid]%P;
                    a[i+j]=(x+y)%P,a[i+j+mid]=(x-y+P)%P;
                }
            }
        }
        if(inv==-1){
            int invn=qpow(tot,P-2);
            for(int i=0;i<tot;i++)a[i]=a[i]*invn%P;
        }
    }
    void init(int n){
        bit=0;
        while((1<<bit)<n)bit++;
        tot=1<<bit;
        rev.resize(tot);
        for(int i=0;i<tot;i++)rev[i]=(rev[i>>1]>>1)|((i&1)<<(bit-1));
    }
    // 多项式乘法
    poly mul(poly a,poly b){
        int n=a.size(),m=b.size();
        init(n+m+1);
        a.resize(tot),b.resize(tot);
        ntt(a,1),ntt(b,1);
        for(int i=0;i<tot;i++)a[i]=a[i]*b[i]%P;
        ntt(a,-1);
        for(int i=n+m;i<tot;i++)a[i]=0;
        return a;
    }
    // 多项式求逆
    poly inv(const poly &f,int n){
        if(n==1)return {qpow(f[0],P-2)};
        auto g=inv(f,n+1>>1);
        static vector<int> c;
        init(2*n+1);
        c.resize(tot);
        g.resize(tot);
        for(int i=0;i<n;i++)c[i]=f[i];
        for(int i=n;i<tot;i++)c[i]=0;
        ntt(c,1),ntt(g,1);
        for(int i=0;i<tot;i++)g[i]=g[i]*(2-c[i]*g[i]%P+P)%P;
        ntt(g,-1);
        for(int i=n;i<tot;i++)g[i]=0;
        return g;
    }
    // 多项式求导
    poly der(const poly &f){
        int n=f.size();
        poly g(n-1);
        for(int i=1;i<n;i++)g[i-1]=f[i]*i%P;
        return g;
    }
    // 多项式求积分
    poly inte(const poly &f){
        int n=f.size();
        poly g(n+1);
        g[0]=0;
        for(int i=1;i<=n;i++)g[i]=f[i-1]*qpow(i,P-2)%P;
        return g;
    }
    // 多项式求值
    int eval(const poly &f,int x){
        int n=f.size(),ans=0;
        for(int i=n-1;i>=0;i--)ans=(ans*x+f[i])%P;
        return ans;
    }
    // 多项式翻转Ar(x)=x^n*A(1/x)
    poly reverse(const poly &f,int n){
        poly g(n);
        for(int i=0;i<n;i++)g[i]=f[n-i-1];
        return g;
    }
    // 多项式加法
    poly add(const poly &f,const poly &g){
        poly c(max(f.size(),g.size()));
        for(int i=0;i<max(f.size(),g.size());i++){
            if(i<f.size())c[i]=f[i];
            if(i<g.size())c[i]=(c[i]+g[i])%P;
        }
        return c;
    }
    // 多项式减法
    poly sub(const poly &f,const poly &g){
        poly c(max(f.size(),g.size()));
        for(int i=0;i<max(f.size(),g.size());i++){
            if(i<f.size())c[i]=f[i];
            if(i<g.size())c[i]=(c[i]-g[i]+P)%P;
        }
        return c;
    }
    // 多项式除法
    pair<poly,poly> div(const poly &f,const poly &g){
        int n=f.size(),m=g.size(),len=n-m+1;
        if(n<m)return {{},f};
        poly fr=reverse(f,n),gr=reverse(g,m);
        gr.resize(len);
        for(int i=len;i<m;i++)gr[i]=0;
        gr=inv(gr,len);
        poly q=mul(fr,gr);
        q=reverse(q,len);
        poly r=sub(f,mul(q,g));
        return {q,r};
    }
    // cipolla
    int cipolla(int x){
        if(qpow(x,P-1>>1)==P-1)return -1;
        while(1){
            int a=rng()%P;
            int w=(a*a%P-x+P)%P;
            auto mul=[&](PII a,PII b){
                return PII{(a.F*b.F%P+w*a.S%P*b.S%P)%P,(a.F*b.S%P+a.S*b.F%P)%P};
            };
            auto pow=[&](PII a,int b){
                PII res{1,0};
                for(;b;b>>=1){
                    if(b&1)res=mul(res,a);
                    a=mul(a,a);
                }
                return res.F;
            };
            if(qpow(w,P-1>>1)==P-1){
                int res=pow(PII{a,1},P+1>>1);
                return min(res,P-res);
            }
        }
    }
    // 多项式开根(a0!=1)
    poly sqrt(const poly &f,int n){
        if(n==1){
            if(f[0]==1)return {1};
            return {cipolla(f[0])};
        }
        auto g=sqrt(f,n+1>>1);
        static vector<int> c;
        static const int inv2=qpow(2,P-2);
        init(2*n+1);
        c.resize(tot);
        g.resize(tot);
        for(int i=0;i<n;i++)c[i]=f[i];
        for(int i=n;i<tot;i++)c[i]=0;
        auto ig=inv(g,n);
        ntt(g,1),ntt(ig,1),ntt(c,1);
        for(int i=0;i<tot;i++)g[i]=(g[i]*g[i]%P+c[i])%P*ig[i]%P*inv2%P;
        ntt(g,-1);
        for(int i=n;i<tot;i++)g[i]=0;
        return g;
    }   
    // 多项式对数ln
    poly ln(const poly &f){
        return inte(mul(der(f),inv(f,f.size())));
    }
    // 多项式指数exp
    poly exp(const poly &f,int n){
        if(n==1)return {1};
        poly g=exp(f,n+1>>1);
        g.resize(n);
        poly lng=ln(g);
        init(2*n+1);
        g.resize(tot);
        lng.resize(tot);
        for(int i=0;i<n;i++)lng[i]=(f[i]-lng[i]+P)%P;
        for(int i=n;i<tot;i++)lng[i]=g[i]=0;
        lng[0]++;
        ntt(lng,1);
        ntt(g,1);
        for(int i=0;i<tot;i++)g[i]=g[i]*lng[i]%P;
        ntt(g,-1);
        for(int i=n;i<tot;i++)g[i]=0;
        return g;
    }
    // 多项式快速幂
    poly qpow(const poly &f,int b){
        int n=f.size();
        poly res{1},a=f;
        res[0]=1;
        init(n<<1);
        res.resize(tot);
        a.resize(tot);
        for(;b;b>>=1){
            ntt(a,1);
            if(b&1){
                ntt(res,1);
                for(int i=0;i<tot;i++)res[i]=res[i]*a[i]%P;
                ntt(res,-1);
                for(int i=n;i<tot;i++)res[i]=0;
            }
            for(int i=0;i<tot;i++)a[i]=a[i]*a[i]%P;
            ntt(a,-1);
            for(int i=n;i<tot;i++)a[i]=0;
        }
        return res;
    }
    // 多项式多点求值
    vector<int> evals(const poly &f,vector<int> &x){
        vector<poly> T;
        int n=f.size(),m=x.size();
        T.assign(m<<2,{});
        function<void(int,int,int)> build=[&](int p,int l,int r){
            if(l==r){
                T[p]={(P-x[l])%P,1};
                return;
            }
            int mid=l+r>>1;
            build(p<<1,l,mid),build(p<<1|1,mid+1,r);
            T[p]=mul(T[p<<1],T[p<<1|1]);
            while(T[p].size()>1&&T[p].back()==0)T[p].pop_back();
        };
        vector<int> res(m);
        function<void(int,int,int,poly)> solve=[&](int p,int l,int r,poly f){
            if(r-l+1<=64){
                for(int i=l;i<=r;i++){
                    res[i]=eval(f,x[i]);
                }
                return;
            }
            auto [q1,r1]=div(f,T[p<<1]);
            auto [q2,r2]=div(f,T[p<<1|1]);
            while(r1.size()>1&&r1.back()==0)r1.pop_back();
            while(r2.size()>1&&r2.back()==0)r2.pop_back();
            int mid=l+r>>1;
            solve(p<<1,l,mid,r1);
            solve(p<<1|1,mid+1,r,r2);
        };
        build(1,0,m-1);
        if(f.size()>=T[1].size()){
            auto [q,r]=div(f,T[1]);
            solve(1,0,m-1,r);
        }
        else solve(1,0,m-1,f);
        return res;
    }
}
```

```cpp
struct mint {

	int x;
	inline mint(int o = 0) { x = o%mod; }
	inline mint & operator = (int o) { return x = o%mod, *this; }

	inline mint & operator -= (mint o) { return (x -= o.x) < 0 && (x += mod), *this; }
```

## 拉格朗日插值
$n+1$个数确定$n$次多项式, $f(k)=\sum_{i=0}^n y_i \prod \frac{k-x_j}{x_i-x_j}$
### O(n^2)版本
```cpp
int lagrange(int k,vector<int> &X,vector<int> &Y){
    int ans=0;
    int n=X.size();
    for(int i=0;i<n;i++){
        int fz=1,fm=1;
        for(int j=0;j<n;j++){
            if(i==j)continue;
            fz=fz*(k-X[j])%mod;
            fm=fm*(X[i]-X[j])%mod;
        }
        ans=(ans+Y[i]*fz%mod*qpow(fm,mod-2)%mod)%mod;
    }
    return (ans+mod)%mod;
}
```
### O(n)版本
* 如果$x$取值连续，那么把$x_i$换成$i$,$f(k)=\sum_{i=0}^ny_i \prod \frac{k-i}{i-j}$
* 记$pre_i=\prod_{j=0}^i k-j,suf_i=\prod_{j=i}^n k-j$
* $f(k)=\sum_{i=0}^ny_i\frac{pre_{i-1}suf_{i+1}}{fac_{i-1}fac_{n-i}}$,$n-i$为奇数时分母取负
```cpp
int lagrange(int k,vector<int> &X,vector<int> &Y,int n){
    int ans=0;
    vector<int> pre(n+10),suf(n+10),ifac(n+10);
    pre[0]=(k-X[0])%mod,suf[n+1]=1;
    for(int i=1;i<=n;i++)pre[i]=pre[i-1]*(k-X[i])%mod;
    for(int i=n;i>=0;i--)suf[i]=suf[i+1]*(k-X[i])%mod;
    ifac[0]=ifac[1]=1;
    for(int i=2;i<=n;i++)ifac[i]=-1ll*mod/i*ifac[mod%i]%mod;
    for(int i=2;i<=n;i++)ifac[i]=ifac[i]*ifac[i-1]%mod;
    for(int i=0;i<=n;i++){
        int tmp=Y[i]*(i==0?1:pre[i-1])%mod*suf[i+1]%mod;
        tmp=tmp*ifac[i]%mod*ifac[n-i]%mod;
        if(n-i&1)tmp=mod-tmp;
        ans=(ans+tmp)%mod;
    }
    return (ans+mod)%mod;
}
```

## 重心拉格朗日插值法
* $g=\prod _{i=0}^nk-x_i,t_i=\frac{y_i}{\prod x_i-x_j}$
* $f(k)=g \sum_{i=0}^n\frac{t_i}{k-x_i}$
每新加入一个点只需要计算$t_i$即可，预处理仍然是$O(n^2)$,但是插入是$O(n)$的


## 多项式乘法逆
给定一个多项式 $F(x)$ ，请求出一个多项式 $G(x)$， 满足 $F(x) * G(x) \equiv 1 \pmod{x^n}$。系数对 $998244353$ 取模。

牛顿迭代法
已知一个复合多项式 $G(F(x)) \equiv 0(\mod x^n)$，假设 $F_0(x) \equiv F(x)(\mod x^{\frac{n}{2}}),F_1(x) \equiv F(x)(\mod x^n)$

泰勒公式在$F_0(x)$展开$G(F(x))$,$G(F(x))=\sum_{i=0}^{\infty}\frac{G^{(i)}F_0(x)}{i!}(F(x)-F_0(x))^i$

$G(F_1(x)) \equiv G(F_0(x))+G'(F_0(x))(F_1(x)-F_0(x))(\mod x^n)$

**$F_1(x)=F_0(x)-\frac{G(F_0(x))}{G'(F_0(x))}$**

将$H(t)=F(x)-\frac{1}{t}$套进去

$G_1(x)=G_0(x)-\frac{F(x)-\frac{1}{G_0(x)}}{\frac{1}{G_0(x)^2}}$

$G_1(x)=G_0(x)(2-G_0(x)F(x))$

## 多项式除法
给定一个 $n$ 次多项式 $F(x)$ 和一个 $m$ 次多项式  $G(x)$ ，请求出多项式 $Q(x)$, $R(x)$，满足以下条件：

 - $Q(x)$ 次数为 $n-m$，$R(x)$ 次数小于 $m$
 - $F(x) = Q(x) * G(x) + R(x)$

考虑一种翻转操作，$A_R(x)=x^nA(\frac{1}{x})$，即$A_{n-i}=A_{i}$

$F(\frac{1}{x})=Q(\frac{1}{x})G(\frac{1}{x})+R(\frac{1}{x})$

$x^nF(\frac{1}{x})=x^{n-m}Q(\frac{1}{x}) * x^mG(\frac{1}{x})+x^{n-m+1}*x^{m-1}R(\frac{1}{x})$

$F_R(x)=Q_R(x)G_R(x)+x^{n-m+1}R_R(x)$

$F_R(x) \equiv Q_R(x)G_R(x)(\mod x^{n-m+1})$

$Q_R(x) \equiv F_R(x)G_R^{-1}(x)(\mod x^{n-m+1})$

最后$R(x)=F(x)-G(x)Q(x)$
```cpp
pair<poly,poly> div(const poly &f,const poly &g){
    int n=f.size(),m=g.size(),len=n-m+1;
    poly fr=reverse(f,n),gr=reverse(g,m);
    gr.resize(len);
    for(int i=len;i<m;i++)gr[i]=0;
    gr=inv(gr,len);
    poly q=mul(fr,gr,n,len);
    q=reverse(q,len);
    poly r=sub(f,mul(q,g,len,m));
    return {q,r};
}
```

```cpp
vector<int> evals(const poly &f,vector<int> &x){
    vector<poly> T;
    int n=f.size(),m=x.size();
    T.assign(m<<2,{});
    function<void(int,int,int)> build=[&](int p,int l,int r){
        if(l==r){
            T[p]={(P-x[l])%P,1};
            return;
        }
        int mid=l+r>>1;
        build(p<<1,l,mid),build(p<<1|1,mid+1,r);
        T[p]=mul(T[p<<1],T[p<<1|1]);
        while(T[p].size()>1&&T[p].back()==0)T[p].pop_back();
    };
    vector<int> res(m);
    function<void(int,int,int,poly)> solve=[&](int p,int l,int r,poly f){
        if(r-l+1<=64){
            for(int i=l;i<=r;i++){
                res[i]=eval(f,x[i]);
            }
            return;
        }
        auto [q1,r1]=div(f,T[p<<1]);
        auto [q2,r2]=div(f,T[p<<1|1]);
        while(r1.size()>1&&r1.back()==0)r1.pop_back();
        while(r2.size()>1&&r2.back()==0)r2.pop_back();
        int mid=l+r>>1;
        solve(p<<1,l,mid,r1);
        solve(p<<1|1,mid+1,r,r2);
    };
    build(1,0,m-1);
    if(f.size()>=T[1].size()){
        auto [q,r]=div(f,T[1]);
        solve(1,0,m-1,r);
    }
    else solve(1,0,m-1,f);
    return res;
}
```

那么$f_{i,j}=f_{i-1,j-k}*g_k$是一个卷积形式，NTT求解(不卡模数的情况下)

## HDU7328 Snake

有$n$个长度为1的蛇，求集合个数(比如$n=2$集合有{1->2},{2->1},{1,2},{2,1}，最终有$m$条蛇，每条蛇长度最多为$k$

容斥做法：

考虑没有排列的因素，$f(i)$表示至少$i$条蛇大于$k$的方案

$ans=f_0-f_1+f_2....$

* 对于$i$，选出$i$个集合等于$k$的，$\binom{m}{i}$
* 剩余$n-ik$分成$m$份，$\binom{n-ik-1}{m-1}$

考虑排序，$\frac{ans \cdot n!}{m!}$全排列，但是集合不排

```cpp
void solve(){
    int n,m,k;
    cin>>n>>m>>k;
    vector<int> f(n+1);
    for(int i=0;i<=m;i++){
        f[i]=comb.binom(m,i)*comb.binom(n-i*k-1,m-1)%mod;
    }
    int ans=0;
    for(int i=0;i<=m;i++){
        if(i&1)ans=(ans+mod-f[i])%mod;
        else ans=(ans+f[i])%mod;
    }
    cout<<ans*comb.fac[n]%mod*comb.inv[m]%mod<<"\n";
}
```

生成函数做法：

每个集合的生成函数$F(x)=x^1+x^2+...+x^k=x \frac{1-x^k}{1-x}$

答案就是$F^m(x)[x^n]$，化简和abc248c一样

最后答案为$\frac{ans \cdot n!}{m!}$

## 2017CCPC 杭州Marriage(容斥+生成函数)

有$n$个家庭，每个家庭有$a_i$个男孩和$b_i$个女孩，$\sum a_i=\sum b_i$，除了同家庭，其余男女配对，问方案数

考虑容斥，$f_i$表示至少$k$对同家庭的

设$p=\min(a_i,b_i)$为当前家庭的贡献，设贡献$i$对，方案为$c_i=\binom{a}{i} \binom{b}{i}i!$，生成函数为$\sum_{i=0}^p c_ix^i$

那么所有生成函数相乘，$f_i=[x^i]$

容斥一下计算答案即可

复杂度$O(n \log^2n)$

分治NTT需要用优先队列启发式合并实现

```cpp
void solve(){
    int n;
    cin>>n;
    vector<int> a(n+1),b(n+1);
    for(int i=1;i<=n;i++)cin>>a[i]>>b[i];
    int s=accumulate(a.begin()+1,a.end(),0LL);
    int t=0;
    for(int i=1;i<=n;i++)t+=min(a[i],b[i]);
    vector<poly> f(n+1);
    for(int i=1;i<=n;i++){
        int p=min(a[i],b[i]);
        f[i].resize(p+1);
        for(int j=0;j<=p;j++)f[i][j]=comb.binom(a[i],j)*comb.binom(b[i],j)%mod*comb.fac[j]%mod;
    }
    auto cmp=[&](int x,int y){
        return f[x].size()>f[y].size();
    };
    priority_queue<int,vector<int>,decltype(cmp)> pq(cmp);
    for(int i=1;i<=n;i++)pq.push(i);
    while(pq.size()>1){
        auto x=pq.top();pq.pop();
        auto y=pq.top();pq.pop();
        int z=f.size();
        auto g=mul(f[x],f[y]);
        g.resize(min((int)g.size(),t+1));
        f.push_back(g);
        pq.push(z);
    }
    auto F=f.back();
    int ans=0;
    for(int i=0;i<=t;i++){
        int tmp=F[i]*comb.fac[s-i]%mod;
        if(i&1)tmp=mod-tmp;
        ans=(ans+tmp)%mod;
    }
    cout<<ans<<'\n';
}
```

## CCPC桂林B

求$A_i=(\sum_{j=i}^{\min(i+L-1,n)}A_j) \mod P$，执行$m$次

将数组$A$倒序，构造数组$G(x)=x^0+x^1+x^2+...+x^L$

每执行一次操作即$A(x)*G(x)$

答案是$A(x)*G^m(x)$

```cpp
void solve(){
    int _=0;
    cin>>_;
    for(int T=1;T<=_;T++){
        int n,L,m;
        cin>>n>>L>>m;
        poly f(n);
        for(int i=0;i<n;i++)cin>>f[i];
        ranges::reverse(f);
        poly g(L,1);
        g=qpow(g,m,n);
        
        f=mul(f,g);
        f.resize(n);
        cout<<"Case "<<T<<": ";
        for(int i=n-1;i>=0;i--)cout<<f[i]<<" ";
        cout<<"\n";
    }
}
```

## 2018沈阳M

多重背包问题，每个位置有数量为$a$价值为$b$的物品

多次在线询问，求$[L,R]$选够价值$c$个物品的方案

每个位置的生成函数为$1+x^b+x^{2b}+...+x^{ab}=\frac{1-x^{(a+1)b}}{1-x^b}$

答案为$\prod_{i=l}^r\frac{1-x^{(a+1)b}}{1-x^b}=\prod_{i=1}^r\frac{x^{(a+1)b}-1}{x^b-1}\prod_{i=1}^{l-1}\frac{x^b-1}{x^{(a+1)b}-1}$

```cpp
void solve(){
    int _;
    cin>>_;
    for(int T=1;T<=_;T++){
        cout<<"Case #"<<T<<":\n";
        int n,m;
        cin>>n>>m;
        const int M=2001;
        vector<int> f(M),g(M);
        vector sf(n+1,vector<int>(M)),sg(n+1,vector<int>(M));
        f[0]=g[0]=sf[0][0]=sg[0][0]=1;
        for(int i=1;i<M;i++)sf[0][i]=1;
        auto _01=[&](vector<int> &dp,int x){
            for(int i=M-1;i>=x;i--)dp[i]=(dp[i]-dp[i-x]+mod)%mod;
        };
        auto _complete=[&](vector<int> &dp,int x){
            for(int i=x;i+x<M;i++)dp[i]=(dp[i]+dp[i-x])%mod;
        };
        for(int i=1;i<=n;i++){
            int a,b;
            cin>>a>>b;
            _01(f,(a+1)*b);
            _01(g,b);
            _complete(f,b);
            _complete(g,(a+1)*b);
            sf[i][0]=f[0];
            sg[i][0]=g[0];
            for(int j=1;j<M;j++){
                sf[i][j]=(sf[i][j-1]+f[j])%mod;
                sg[i][j]=g[j];
            }
        }
        int ans=0;
        while(m--){
            int l,r,c;
            cin>>l>>r>>c;
            l=(l+ans)%n+1,r=(r+ans)%n+1;
            if(l>r)swap(l,r);
            ans=0;
            for(int i=0;i<=c;i++)ans=(ans+sf[r][i]*sg[l-1][c-i]%mod)%mod;
            cout<<ans<<"\n";
        }
    }
}
```
